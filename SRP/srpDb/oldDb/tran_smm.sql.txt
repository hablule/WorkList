CREATE VIEW `TRANSACTION_SUMMARY` AS
SELECT T_TRANSACTION_ID,DOCUMENTNO,TRXDATE,T_TRANSACTION.M_BPARTNER_ID,M_BPARTNER.NAME,ISSALESTRX,SUBTRXAMOUNT,TRXTAXAMOUNT,GRANDTRXAMOUNT,DOCSTATUS,PROCESSED
FROM T_TRANSACTION LEFT JOIN M_BPARTNER
ON T_TRANSACTION.M_BPARTNER_ID= M_BPARTNER.M_BPARTNER_ID

CREATE VIEW `TRANSACTION_dtl` AS
SELECT T_TRXDETAIL_ID,T_TRXDETAIL.T_TRANSACTION_ID,DOCUMENTNO,T_TRANSACTION.TRXDATE,T_TRANSACTION.M_BPARTNER_ID,M_BPARTNER.NAME,T_TRANSACTION.ISSALESTRX,T_TRXDETAIL.M_PRODUCT_ID,M_PRODUCT.CODE,M_PRODUCT.NAME as PRODUCT,T_TRXDETAIL.M_UOM_ID,M_UOM.ABBREVATION,TRXQUANTITY,UNITPRICE,DISCOUNTAMT,LINENETAMT,LINETAXAMOUNT,(LINENETAMT + LINETAXAMOUNT) as LINETOTAL,DOCSTATUS,PROCESSED
FROM (((T_TRXDETAIL LEFT JOIN T_TRANSACTION
ON T_TRXDETAIL.T_TRANSACTION_ID = T_TRANSACTION.T_TRANSACTION_ID) LEFT JOIN M_BPARTNER
ON T_TRANSACTION.M_BPARTNER_ID= M_BPARTNER.M_BPARTNER_ID) LEFT JOIN M_PRODUCT
ON T_TRXDETAIL.M_PRODUCT_ID = M_PRODUCT.M_PRODUCT_ID) LEFT JOIN M_UOM
ON T_TRXDETAIL.M_UOM_ID = M_UOM.M_UOM_ID
